# 快速开始：创建你的第一个智能代理

from lingwei import Agent

# 初始化灵薇代理
agent = Agent(
  model="minimax/MiniMax-M2.5",
  platforms=["feishu", "telegram"],
  dgm_enabled=True
)

# 启动自我进化模式
agent.enable_dgm(
  improvement_interval=3600,
  self_modify=True
)

# 让它开始工作
agent.run()
